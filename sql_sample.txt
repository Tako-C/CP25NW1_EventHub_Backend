-- หาตามช่วงอายุ
SELECT 
    ue.EVENT_ID,
    CASE 
        WHEN TRUNC(MONTHS_BETWEEN(SYSDATE, u.DATE_OF_BIRTH) / 12) < 18 THEN 'ต่ำกว่า 18 ปี'
        WHEN TRUNC(MONTHS_BETWEEN(SYSDATE, u.DATE_OF_BIRTH) / 12) BETWEEN 18 AND 24 THEN '18 - 24 ปี'
        WHEN TRUNC(MONTHS_BETWEEN(SYSDATE, u.DATE_OF_BIRTH) / 12) BETWEEN 25 AND 34 THEN '25 - 34 ปี'
        WHEN TRUNC(MONTHS_BETWEEN(SYSDATE, u.DATE_OF_BIRTH) / 12) BETWEEN 35 AND 44 THEN '35 - 44 ปี'
        ELSE '45 ปีขึ้นไป'
    END AS AGE_RANGE,
    COUNT(ue.USER_ID) AS TOTAL_PARTICIPANTS
FROM USER_EVENTS ue
JOIN USERS u ON ue.USER_ID = u.ID
WHERE u.DATE_OF_BIRTH IS NOT NULL
GROUP BY 
    ue.EVENT_ID,
    CASE 
        WHEN TRUNC(MONTHS_BETWEEN(SYSDATE, u.DATE_OF_BIRTH) / 12) < 18 THEN 'ต่ำกว่า 18 ปี'
        WHEN TRUNC(MONTHS_BETWEEN(SYSDATE, u.DATE_OF_BIRTH) / 12) BETWEEN 18 AND 24 THEN '18 - 24 ปี'
        WHEN TRUNC(MONTHS_BETWEEN(SYSDATE, u.DATE_OF_BIRTH) / 12) BETWEEN 25 AND 34 THEN '25 - 34 ปี'
        WHEN TRUNC(MONTHS_BETWEEN(SYSDATE, u.DATE_OF_BIRTH) / 12) BETWEEN 35 AND 44 THEN '35 - 44 ปี'
        ELSE '45 ปีขึ้นไป'
    END
ORDER BY ue.EVENT_ID, AGE_RANGE;

-- หาตามเพศ
SELECT 
    EVENT_ID, 
    GENDER_CODE, 
    GENDER_NAME, 
    COUNT(*) AS TOTAL_USERS
FROM (
    SELECT 
        ue.EVENT_ID,
        NVL(u.GENDER, 'N') AS GENDER_CODE,
        CASE 
            WHEN u.GENDER = 'M' THEN 'ชาย'
            WHEN u.GENDER = 'F' THEN 'หญิง'
            WHEN u.GENDER = 'U' THEN 'ไม่ระบุ'
            ELSE 'อื่นๆ/ไม่ระบุ'
        END AS GENDER_NAME
    FROM USER_EVENTS ue
    JOIN USERS u ON ue.USER_ID = u.ID
)
GROUP BY EVENT_ID, GENDER_CODE, GENDER_NAME
ORDER BY EVENT_ID, GENDER_CODE



-- หาตามอาชีพ
SELECT 
    ue.EVENT_ID,
    j.NAME_TH AS JOB_NAME,
    COUNT(ue.USER_ID) AS TOTAL_PARTICIPANTS
FROM USER_EVENTS ue
JOIN USERS u ON ue.USER_ID = u.ID
JOIN JOBS j ON u.JOB_ID = j.ID
GROUP BY ue.EVENT_ID, j.NAME_TH
ORDER BY ue.EVENT_ID, TOTAL_PARTICIPANTS DESC;

-- หาตามจังหวัด
SELECT 
    ue.EVENT_ID,
    c.NAME_TH AS CITY_NAME,
    COUNT(ue.USER_ID) AS TOTAL_PARTICIPANTS
FROM USER_EVENTS ue
JOIN USERS u ON ue.USER_ID = u.ID
JOIN CITIES c ON u.CITIES_ID = c.ID
GROUP BY ue.EVENT_ID, c.NAME_TH
ORDER BY ue.EVENT_ID, TOTAL_PARTICIPANTS DESC;
